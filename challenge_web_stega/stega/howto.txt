zip starwars.zip starwars.txt
cat chess.jpg starwars.zip >cat.jpg
apt install steghide 
steghide embed -cf magical.jpg -ef cat.jpg -sf full.jpg -p "magical"
-> magical (password)
(extraction)
steghide extract -sf full.jpg -xf extract.jpg -p "magical"
unzip extract.jpg
